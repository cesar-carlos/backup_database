void main(List<String> args) async {
  WidgetsFlutterBinding.ensureInitialized();
  await dotenv.load(fileName: '.env');

  // Detect mode from command line or config file
  final appMode = detectAppMode(args);

  // Log mode
  LoggerService.info('Starting Backup Database in ${appMode.name} mode');

  // Configure dependencies based on mode
  setupDependencies(appMode);

  runApp(MyApp(appMode: appMode));

AppMode detectAppMode(List<String> args) {
  // 1. Check command line args (highest priority)
  if (args.contains('--mode=server')) return AppMode.server;
  if (args.contains('--mode=client')) return AppMode.client;

  // 2. Check config file (created by installer)
  try {
    final configFile = File('config/mode.ini');
    if (configFile.existsSync()) {
      final contents = configFile.readAsStringSync();
      if (contents.contains('Type=Server')) return AppMode.server;
      if (contents.contains('Type=Client')) return AppMode.client;
    }
  } catch (e) {
    LoggerService.warning('Could not read mode.ini: $e');
  }

  // 3. Check environment variable
  final modeFromEnv = dotenv.env['APP_MODE'];
  if (modeFromEnv?.toLowerCase() == 'server') return AppMode.server;
  if (modeFromEnv?.toLowerCase() == 'client') return AppMode.client;

  // 4. Default to server (backward compatibility)
  LoggerService.info('No mode specified, defaulting to Server mode');
  return AppMode.server;
}

        useMaterial3: true,
      ),
      initialRoute: appMode == AppMode.server
          ? Routes.serverDashboard
          : Routes.clientLogin,
      onGenerateRoute: (settings) {
        // Route baseada no modo
        if (appMode == AppMode.server) {
          return _generateServerRoute(settings);
        } else {
          return _generateClientRoute(settings);
        }
      },
    );
  }

    switch (settings.name) {
      case Routes.serverDashboard:

          builder: (_) => const ServerDashboardPage(),

      case Routes.connectedClients:
        return MaterialPageRoute(
          builder: (_) => const ConnectedClientsPage(),
        );
      case Routes.serverSettings:
        return MaterialPageRoute(
          builder: (_) => const ServerSettingsPage(),
        );
      default:
        return MaterialPageRoute(
          builder: (_) => const ServerDashboardPage(),
        );
    }
